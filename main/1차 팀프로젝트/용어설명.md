
```python
# 📦 라이브러리(Library)란?
# - 파이썬 기능들을 미리 만들어둔 '도구 상자'
# - 반복적으로 필요한 기능을 쉽게 가져다 쓸 수 있음
# - 설치만 하면 바로 불러와서 사용 가능

# 예:
# pip install pandas matplotlib

# =========================================

# 🧰 pandas (판다스)
# - 데이터를 표처럼 다루는 라이브러리 (엑셀처럼 행/열 기반)
# - CSV, Excel 파일 읽고 쓰기 가능
# - 데이터 분석, 통계 작업에 자주 쓰임

import pandas as pd  # pandas를 pd라는 별명(alias)으로 불러옴

data = {'이름': ['철수', '영희'], '나이': [20, 22]}
df = pd.DataFrame(data)
print(df)

# 결과:
#    이름  나이
# 0  철수   20
# 1  영희   22

# =========================================

# 📊 matplotlib (맷플롯립)
# - 데이터를 시각화(그래프)로 보여주는 라이브러리
# - 선 그래프, 막대 그래프, 파이 차트 등 지원
# - 분석 결과를 눈에 잘 보이게 표현할 때 사용

import matplotlib.pyplot as plt  # matplotlib.pyplot을 plt로 줄여 부름

x = [1, 2, 3, 4]
y = [10, 20, 25, 30]

plt.plot(x, y)
plt.title("샘플 그래프")
plt.xlabel("X 값")
plt.ylabel("Y 값")
plt.show()

# =========================================

# 🆔 alias (별명) 관습 정리
# - pandas → pd
# - numpy → np
# - matplotlib.pyplot → plt

# 별명(alias)은 그냥 짧게 쓰려고 붙이는 이름이며, 대부분의 개발자가 사용하는 공통 약속임
```
